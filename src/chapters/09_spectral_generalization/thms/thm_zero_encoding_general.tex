\begin{theorem}[Spectral Encoding of Nontrivial Zeros of \texorpdfstring{$L(s, \pi)$}{L(s, pi)}]
\label{thm:zero_encoding_general}
Let \( \pi \in \mathcal{A}_{\mathrm{cusp}}(\mathrm{GL}_n) \), and assume the analytic properties of \( \Xi(s, \pi) \) from \defref{def:analytic_properties_XiPi}, the exponential kernel decay condition in \defref{def:kernel_decay_condition}, and the determinant identity of \thmref{thm:spectral_determinant_identity_Lpi}.

Then the eigenvalues of the operator \( \Lsympi \in \TC(\Hpsipi) \) encode the nontrivial zeros of \( L(s, \pi) \). Specifically:
\[
\Spec(\Lsympi) = \left\{ \mu_\rho := \frac{1}{i}(\rho - \tfrac{1}{2}) : \Xi(\rho, \pi) = 0 \right\},
\]
with each eigenvalue \( \mu_\rho \in \C \) appearing with multiplicity equal to the order of vanishing of \( \Xi(s, \pi) \) at \( s = \rho \).
\end{theorem}
