\subsection*{Summary}

\begin{itemize}
  % --- Definitions ---
  \item \defref{def:heat_operator} — Definition of the heat semigroup \( e^{-tL^2} \) for compact, self-adjoint operators: strongly continuous and trace class for all \( t > 0 \).

  % --- Kernel Bounds and Local Expansion ---
  \item \lemref{lem:hk_upper_bound} — Upper bound:
  \[
  \Tr(e^{-t\Lsym^2}) \leq c_2\, t^{-1/2},
  \]
  via spectral partitioning and Schatten norm estimates.

  \item \lemref{lem:hk_lower_bound} — Lower bound:
  \[
  \Tr(e^{-t\Lsym^2}) \geq c_1\, t^{-1/2},
  \]
  confirming sharpness of the leading singular growth.

  \item \lemref{lem:hk_expansion_uniform} — Uniform short-time diagonal kernel expansion:
  \[
  K_t(x,x) \sim \sum_{n=0}^\infty a_n(x)\, t^{n - \frac{1}{2}}, \qquad a_n(x) \in C^\infty(\R) \cap \Schwartz(\R).
  \]

  % --- Global Trace Expansion ---
  \item \lemref{lem:heat_trace_expansion} — Trace asymptotics:
  \[
  \Tr(e^{-t\Lsym^2}) \sim \sum_{n=0}^\infty A_n\, t^{n - \frac{1}{2}}, \qquad
  A_n := \int_{\R} a_n(x)\, dx.
  \]

  \item \propref{prop:two_sided_heat_trace_bounds} — Two-sided bound:
  \[
  c_1\, t^{-1/2} \leq \Tr(e^{-t\Lsym^2}) \leq c_2\, t^{-1/2},
  \]
  uniformly for \( t \in (0, t_0] \), establishing regular variation.

  \item \propref{prop:heat_trace_uniform_conv} — Global expansion:
  \[
  \Tr(e^{-t\Lsym^2}) \sim \sum_{n=0}^\infty A_n\, t^{n - \frac{1}{2}},
  \]
  with termwise integrability and exponentially decaying tail.

  \item \propref{prop:short_time_heat_expansion} — Refined singular expansion:
  \[
  \Tr(e^{-t\Lsym^2})
  = \frac{1}{\sqrt{4\pi t}} \log\left( \frac{1}{t} \right)
  + c_0 \sqrt{t} + o(\sqrt{t}), \qquad c_0 := \int_{\R} a_1(x)\, dx.
  \]
  The logarithmic term governs determinant growth and spectral counting.

  % --- Spectral and Determinant Linkages ---
  \item \lemref{lem:log_derivative_determinant} — Logarithmic derivative of the spectral determinant:
  \[
  \log \detz(\Lsym^2)
  = -\int_0^\infty \frac{\Tr(e^{-t \Lsym^2}) - P(t)}{t} \, dt,
  \]
  where \( P(t) \sim \frac{1}{\sqrt{4\pi t}} \log(1/t) + \dots \) is the parametrix subtraction.

  \item \remref{rem:spectral_dimension} — Spectral dimension and Weyl-type asymptotics:
  \[
  N(\lambda) \sim C \lambda^{1/2} \log \lambda,
  \]
  identifying log-corrected spectral density for \( \Lsym^2 \).
\end{itemize}
