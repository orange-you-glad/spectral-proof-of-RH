\begin{theorem}[Zeta-Regularized Determinant Identity for \texorpdfstring{$\Xi(s,\pi)$}{Xi(s, pi)}]
\label{thm:spectral_determinant_identity_Lpi}
Let \( \pi \in \mathcal{A}_{\mathrm{cusp}}(\mathrm{GL}_n) \), and assume the analytic properties of \( \Xi(s, \pi) \) as in \defref{def:analytic_properties_XiPi}, and the kernel decay condition in \defref{def:kernel_decay_condition}.

Let \( \Lsympi \in \TC(\Hpsipi) \) denote the limit operator
\[
\Lsympi := \lim_{t \to 0^+} \Ltpi,
\]
where \( \Ltpi \) is the mollified convolution operator defined in \defref{def:convolution_operator_Ltpi}. Then the following identity holds:
\[
\detz(I - \lambda \Lsympi) = \frac{\Xi\left(\tfrac{1}{2} + i\lambda, \pi\right)}{\Xi\left(\tfrac{1}{2}, \pi\right)},
\qquad \forall \lambda \in \C.
\]
Here \( \detz \) denotes the Carleman zeta-regularized Fredholm determinant on \( \TC(\Hpsipi) \).

\end{theorem}
