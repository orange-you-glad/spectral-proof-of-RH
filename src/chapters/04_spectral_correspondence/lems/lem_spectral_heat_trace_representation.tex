\begin{lemma}[Spectral Heat Trace Representation]
\label{lem:spectral_heat_trace_representation}
Let \( L_{\mathrm{sym}} \in \mathcal{C}_1(H_{\Psi_\alpha}) \) be a compact, self-adjoint operator with discrete spectrum \( \{\mu_\rho\} \subset \mathbb{R} \). Then the associated spectral projection measure \( E_\lambda \) satisfies:
\[
\operatorname{Tr}(e^{-tL_{\mathrm{sym}}^2}) = \int_{\mathbb{R}} e^{-t\lambda^2} \, dN(\lambda),
\]
where the eigenvalue counting function \( N(\lambda) \) is given by
\[
N(\lambda) := \sum_{\mu_\rho \leq \lambda} \operatorname{mult}(\mu_\rho),
\]
counting each eigenvalue \( \mu_\rho \) with its multiplicity.

Consequently, the canonical Carleman zeta-regularized Fredholm determinant satisfies
\[
\log \det\nolimits_{\zeta}(I - \lambda L_{\mathrm{sym}}) = - \int_{\mathbb{R}} \log\left(1 - \frac{\lambda}{\mu_\rho} \right) dN(\mu_\rho),
\]
valid for all \( \lambda \in \mathbb{C} \setminus \{\mu_\rho\} \).
\end{lemma}
