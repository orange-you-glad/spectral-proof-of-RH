\begin{lemma}[Trace-Norm Convergence of \texorpdfstring{$\Ltpi$}{Ltpi}]
\label{lem:trace_class_convergence_general}
Let \( \pi \in \mathcal{A}_{\mathrm{cusp}}(\mathrm{GL}_n) \), and suppose the kernel \( k_t^{(\pi)} \) satisfies the exponential decay condition in \defref{def:kernel_decay_condition} for some \( \alphapi > \pi_\pi \). Then:
\begin{enumerate}
  \item For all \( t > 0 \), the convolution operator \( \Ltpi \in \TC(\Hpsipi) \), as defined in \defref{def:convolution_operator_Ltpi}.
  \item The family \( \{ \Ltpi \}_{t > 0} \subset \TC(\Hpsipi) \) is uniformly bounded in the trace norm.
  \item The limit
  \[
  \Lsympi := \lim_{t \to 0^+} \Ltpi
  \]
  exists in \( \TC(\Hpsipi) \), and defines a compact, self-adjoint, trace-class operator.
\end{enumerate}
\end{lemma}
