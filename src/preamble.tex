%==============================
% Wide Check (for Inverse FT)
%==============================
\makeatletter
\DeclareRobustCommand{\widecheck}[1]{
  \mathpalette\@widecheck{#1}
}
\def\@widecheck#1#2{%
  \setbox\z@\hbox{\m@th#1#2}%
  \setbox\tw@\hbox{\m@th#1\widehat{\box\z@}}%
  \setbox\tw@\hbox to \wd\z@{\hss\box\tw@\hss}%
  {\vbox{\offinterlineskip
      \box\tw@
      \vskip -\ht\z@
      \box\z@}}}
\makeatother
