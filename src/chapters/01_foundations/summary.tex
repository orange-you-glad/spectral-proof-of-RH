\subsection*{Summary}
\label{sec:foundations_summary}

\textbf{Operator-Theoretic Foundations}
\begin{itemize}
  \item \defref{def:compact_operator} — Compact operators: norm limits of finite-rank maps with discrete spectrum.
  \item \defref{def:trace_class_operator}, \defref{def:trace_norm} — Trace-class operators \( T \in \TC(H) \) with finite trace norm \( \|T\|_{\Tr} := \Tr(|T|) \); Banach structure and unitary invariance.
  \item \defref{def:selfadjoint_operator} — Self-adjointness as maximal symmetry enabling spectral calculus and semigroup evolution.
\end{itemize}

\textbf{Weighted Spaces and Function Classes}
\begin{itemize}
  \item \defref{def:exponential_weight}, \defref{def:weighted_schwartz_space} — The space \( \HPsi := L^2(\R, e^{\alpha|x|}\,dx) \), with \( \Schwartz(\R) \subset \HPsi \) as a dense analytic core.
  \item \lemref{lem:density_schwartz_weighted_L2} — Density of \( \Schwartz \subset \HPsi \) in norm and graph topology.
  \item \remref{rem:sobolev_core_reference} — Alternate justification: \( \Schwartz \hookrightarrow H^s_\alpha \hookrightarrow \HPsi \) via Sobolev embeddings.
\end{itemize}

\textbf{Analytic and Spectral Estimates}
\begin{itemize}
  \item \lemref{lem:xi_growth_bound}, \lemref{lem:weighted_L1_inverse_FT_xi} — The profile \( \Xi(\tfrac{1}{2} + i\lambda) \in \PW{\pi} \), with inverse transform in \( L^1(\R, \Psi_\alpha^{-1}) \).
  \item \lemref{lem:decay_mollified_kernel}, \lemref{lem:L1_integrability_conjugated_kernel}, \lemref{lem:mollified_profile_decay} — Mollifiers \( k_t \in \Schwartz \), exhibiting rapid exponential decay and integrability.
  \item \lemref{lem:uniform_L1_conjugated_kernel}, \lemref{lem:trace_class_via_weighted_L1}, \lemref{lem:trace_norm_convergence_Lt_to_Lsym} — Trace-norm convergence \( \|L_t - \Lsym\|_{\TC} \to 0 \); Simon’s trace-class inclusion criterion.
  \item \lemref{lem:trace_class_conjugated_kernel}, \lemref{lem:trace_class_failure_alpha_leq_pi}, \propref{prop:trace_class_sharpness} — \textbf{Sharp threshold:} trace-class fails for \( \alpha \le \pi \); suffices if and only if \( \alpha > \pi \).
  \item \lemref{lem:unitary_conjugation_trace_class}, \lemref{lem:kernel_L2_weighted_bound} — Trace norm preserved under unitary weight conjugation; estimates in weighted \( L^2 \) spaces.
\end{itemize}

\textbf{Operator Properties of \texorpdfstring{\( L_t \)}{Lt}}
\begin{itemize}
  \item \propref{prop:boundedness_Lt_weighted}, \propref{prop:compactness_Lt} — Boundedness and compactness of \( L_t \) via mollified kernel structure.
  \item \propref{prop:symmetry_Lt_Schwartz}, \propref{prop:selfadjointness_Lt} — \( L_t \) is symmetric on \( \Schwartz \) and extends to a self-adjoint operator on \( \HPsi \).
  \item \propref{prop:core_schwartz_density} — \( \Schwartz \) is a core domain for the limit operator \( \Lsym \).
\end{itemize}

\textbf{Canonical Operator Realization}
\begin{itemize}
  \item \thmref{thm:canonical_operator_realization} — Convergence \( L_t \to \Lsym \in \TC(\HPsi) \); defines the canonical compact, self-adjoint operator realizing the spectral determinant identity.
\end{itemize}

\paragraph{Chapter Closure.}
This chapter constructs the analytic and operator-theoretic foundation for the spectral encoding of the Riemann Hypothesis. The canonical convolution operator \( \Lsym \in \TC(\HPsi) \) is defined as the trace-norm limit of mollified Fourier convolution operators \( L_t \), as established in \lemref{lem:trace_norm_convergence_Lt_to_Lsym}. Its analytic structure depends on Paley–Wiener bounds, exponential kernel decay (\lemref{lem:mollified_profile_decay}), trace embedding theorems, and Sobolev core regularity. A sharp boundary emerges: \( \alpha > \pi \) is both necessary and sufficient for trace-class regularity, as shown in \lemref{lem:trace_class_failure_alpha_leq_pi}. These constructions culminate in the determinant identity
\[
\detz(I - \lambda \Lsym)
= \frac{\Xi\left(\tfrac{1}{2} + i\lambda \right)}{\Xi\left(\tfrac{1}{2} \right)},
\]
proven in \thmref{thm:det_identity_revised}, which anchors the spectral formulation of RH developed in later chapters.
